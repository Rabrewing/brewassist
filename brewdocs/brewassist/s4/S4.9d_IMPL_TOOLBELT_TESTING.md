#### 2.3. Wire MCP Buttons → Toolbelt

In `WorkspaceSidebarLeft.tsx` (or wherever those left-panel buttons are wired):

For each MCP tool:

* Identify the onClick handler. Example:

```tsx
<button onClick={() => handleFileAssistantClick()}>
  File Assistant
</button>
```

Wrap each handler with a Toolbelt gate:

```tsx
import { canRunTool, type ToolbeltToolId } from '@/lib/toolbeltConfig';

function tryRunTool(toolId: ToolbeltToolId, mode: BrewMode, currentTier: 1 | 2 | 3) {
  if (!canRunTool({ toolId, mode, currentTier })) {
    // TODO: replace with your toast/hud system
    alert('This action requires a higher Toolbelt tier or is not allowed in this mode.');
    return;
  }

  // Existing behavior
  runTool(toolId);
}
```

Then your button becomes:

```tsx
<button onClick={() => tryRunTool('file-assistant', currentMode, currentTier)}>
  File Assistant
</button>
```

Do this for:

* `file-assistant`
* `git-command-center`
* `database-assistant`
* `research-nims`
* `suggest-edits`

**Result:** If the user is in HRM / Tier 1 and clicks “File Assistant,” nothing dangerous happens — they just get a “Tier required” message.

### 3. Test Plan — S4.9d Toolbelt

#### 3.1. Unit Tests — `toolbeltConfig`

Goal: prove the policy math is correct.

```ts
import {
  canRunTool,
  TOOLBELT_MODE_MATRIX,
  TOOLBELT_TOOL_CONFIG,
} from '@/lib/toolbeltConfig';

describe('toolbeltConfig', () => {
  test('HRM only allows Tier 1', () => {
    expect(TOOLBELT_MODE_MATRIX.hrm).toEqual([1]);
  });

  test('Agent allows all tiers', () => {
    expect(TOOLBELT_MODE_MATRIX.agent).toEqual([1, 2, 3]);
  });

  test('Tier 1 cannot run file-assistant', () => {
    const allowed = canRunTool({
      toolId: 'file-assistant',
      mode: 'llm',
      currentTier: 1,
    });
    expect(allowed).toBe(false);
  });

  test('Tier 2 can run file-assistant in LLM', () => {
    const allowed = canRunTool({
      toolId: 'file-assistant',
      mode: 'llm',
      currentTier: 2,
    });
    expect(allowed).toBe(true);
  });

  test('HRM cannot run git-command-center even at Tier 3', () => {
    const allowed = canRunTool({
      toolId: 'git-command-center',
      mode: 'hrm',
      currentTier: 3,
    });
    expect(allowed).toBe(false);
  });
});
```

**Acceptance:**

* All toolbeltConfig tests pass with `pnpm test`.
